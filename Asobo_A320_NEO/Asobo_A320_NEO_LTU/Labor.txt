#include <stdio.h>
int main()
{
	float gewicht = 0.0f, groesse = 0.0f, bmi = 0.0f;
	scanf("%f", &gewicht);
	scanf("%f", &groesse);
	bmi = ((gewicht / 1000) / ((groesse / 100) * (groesse / 100)));
	if (bmi < 18.4) {
		printf("Untergewicht %.1f", bmi);
	}
	else {
		if (bmi < 24.9) {
			printf("Normalgewicht %.1f", bmi);
		}
		else {
			printf("Uebergewicht %.1f", bmi);
		}
	}
	return 0;
}

#include <stdio.h>
int main()
{
	float w = 0.0f, r = 0.0f,v = 0.0, phi = 0.8f, a = 0.0f, m = 0.0f, e = 0.0f;
	char geschlecht;
	scanf("%c", &geschlecht);
	
	scanf("%f", &m);
	scanf("%f", &v);
	scanf("%f", &e);

	if (geschlecht == 'm') {
		r = 0.7f;
	}
	else {
		if (geschlecht == 'w') {
			r = 0.6f;
		}
		else {
			r = 0.8f;
		}
	}
	e *= 0.01;
	a = v * e * phi;
	w = (a / (m * r));
	printf("%.2f Promille", w);
	return 0;
}

    unsigned int note, bestanden;
    printf("\nGeben Sie bitte eine Note (1-5) ein: ");
    scanf("%u", &note);
    switch (note) {
    case SPITZE:
    case GUT:
    case BEFRIEDIGEND:
    case AUSREICHEND:
         {
        bestanden = JA;
        break;
            }
    default:
         {
        bestanden = NEIN;
        break;
         }
    }

    return 0;
}
